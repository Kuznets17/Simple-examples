#  аждое значение заканчиваетс€ символом новой строки.
# ћагазин работает 8 часов в день.

# Ќа момент запуска приложени€ все значени€ уже наход€тс€ в файлах.
# Ќаписать программу, котора€ по данным замеров определ€ет интервал времени, когда в магазине было наибольшее количество посетителей за день.
# јргумент программы - путь к каталогу с файлами. ¬ каталоге будут 5 файлов: Cash1, Cash2 ... Cash5.
# ¬ыведите номер интервала, в котором было наибольшее число посетителей в очеред€х магазина на всех кассах.
# ѕервый интервал идет под номером 1, последний под номером 16. 
# ¬ случае обнаружени€ нескольких интервалов следует выводить первый из них.

import sys
import argparse as argp

# получаем им€ директории, содержащей файлы с входными даными из командной строки
def createParser ():
    parser = argp.ArgumentParser()
    parser.add_argument ('-d', '--dir', required=True)
    return parser

if __name__ == '__main__':
    parser = createParser()
    namespace = parser.parse_args(sys.argv[1:])
    
    # помещаем содержимое файлов (с заведомо известными названи€ми) построчно в списки
    fData1 = [float(line.rstrip('\n')) for line in open(namespace.dir + "/Cash1")]
    fData2 = [float(line.rstrip('\n')) for line in open(namespace.dir + "/Cash2")]
    fData3 = [float(line.rstrip('\n')) for line in open(namespace.dir + "/Cash3")]
    fData4 = [float(line.rstrip('\n')) for line in open(namespace.dir + "/Cash4")]
    fData5 = [float(line.rstrip('\n')) for line in open(namespace.dir + "/Cash5")]
    
    # собираем новый список с суммами соответствующих элементов списков входных данных
    # из услови€ задачи следует, что всего таких интервалов 16, так что количество строк в файлах провер€ть не нужно
    dataSum = []
    count = 0
    while (count<16):
        dataSumTemp = fData1[count] + fData2[count] + fData3[count] + fData4[count] + fData5[count]
        dataSum.append(dataSumTemp)
        count += 1
    
    # выводим номер элемента с максимальным значением
    print(dataSum.index(max(dataSum)) + 1)
